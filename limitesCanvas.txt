        if (e.key == "ArrowUp" || e.keyCode == "87") { 
            for (const rect of rects) {
                for (const rect of rects) {
                    if (xChar == rect.xRect && yChar == rect.yRect + L) {
                        isTouching = true
                        idRect = rect.id
                    }
                }
                if (isTouching && rect.id == idRect) {
                    yChar -= L
                    rect.yRect -= L
                    break
                } else if (idRect == 0) {
                    yChar -= L
                    break
                }
            }
        }
